2. Create a new inner component `WorkspaceGate` (rendered inside WorkspaceProvider, replacing direct AppContent):
```jsx
function WorkspaceGate() {
  const { workspaces, loading } = useWorkspace();

  if (loading) {
    // Show same spinner as AuthGate loading state
    return <div className="auth-loading"><span className="spinner" /></div>;
  }

  if (workspaces.length === 0) {
    return <WorkspaceOnboarding onComplete={() => {
      // WorkspaceProvider will re-fetch automatically if we trigger a re-render.
      // Simplest approach: pass a reload callback that forces WorkspaceProvider to
      // re-query. However, WorkspaceProvider already derives state from user and
      // Supabase — the cleanest solution is to have WorkspaceProvider expose a
      // `reloadWorkspaces` function in context that re-runs the query.
      //
      // Implementation: call window.location.reload() as the simplest correct
      // option — after insert the page refreshes, WorkspaceProvider re-fetches,
      // and the user lands on the gallery.
      window.location.reload();
    }} />;
  }

  return <AppContent />;
}
```

3. Update App() to use WorkspaceGate:
```jsx
export default function App() {
  return (
    <AuthProvider>
      <AuthGate>
        <WorkspaceProvider>
          <WorkspaceGate />
        </WorkspaceProvider>
      </AuthGate>
    </AuthProvider>
  );
}
```

Note: `window.location.reload()` is intentional here — it is the simplest correct solution for forcing WorkspaceProvider to re-query after the user creates/joins a workspace. Do NOT attempt a more complex state-lifting approach.
