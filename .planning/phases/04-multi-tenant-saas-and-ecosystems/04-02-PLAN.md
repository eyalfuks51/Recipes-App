  <files>supabase/migrations/20260306_workspaces.sql</files>
  <action>
    Create the file `supabase/migrations/20260306_workspaces.sql` (create `supabase/migrations/` directory if it doesn't exist) with the following SQL that must be run in Supabase Dashboard → SQL Editor:

    ```sql
    -- Workspaces (ecosystems)
    CREATE TABLE IF NOT EXISTS workspaces (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      name TEXT NOT NULL,
      created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    );

    -- Workspace membership (many-to-many: auth.users <-> workspaces)
    CREATE TABLE IF NOT EXISTS workspace_users (
      workspace_id UUID NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
      user_id UUID NOT NULL,  -- references auth.users(id)
      role TEXT NOT NULL DEFAULT 'member',  -- 'owner' | 'member'
      joined_at TIMESTAMPTZ NOT NULL DEFAULT now(),
      PRIMARY KEY (workspace_id, user_id)
    );

    -- Add workspace_id to recipes (nullable for backward compatibility)
    ALTER TABLE recipes
      ADD COLUMN IF NOT EXISTS workspace_id UUID REFERENCES workspaces(id) ON DELETE SET NULL;

    -- Index for efficient workspace recipe queries
    CREATE INDEX IF NOT EXISTS idx_recipes_workspace_id ON recipes(workspace_id);

    -- Add instructions column for Phase 4 Full Recipe View
    ALTER TABLE recipes
      ADD COLUMN IF NOT EXISTS instructions TEXT;

    -- Ingredient checked state per workspace (for stateful checkboxes)
    CREATE TABLE IF NOT EXISTS workspace_ingredient_checks (
      workspace_id UUID NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
      recipe_id UUID NOT NULL REFERENCES recipes(id) ON DELETE CASCADE,
      ingredient_id UUID NOT NULL REFERENCES ingredients(id) ON DELETE CASCADE,
      checked BOOLEAN NOT NULL DEFAULT false,
      updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
      PRIMARY KEY (workspace_id, recipe_id, ingredient_id)
    );
    ```

    Note: The `instructions` column and `workspace_ingredient_checks` table are added here because they are needed by plan 04-05 (Full Recipe View) which runs in Wave 2. Adding them in this single migration avoids a second schema change.

    Also create a `supabase/README.md` explaining how to apply migrations manually via Supabase Dashboard SQL Editor (since there is no Supabase CLI in this project).
